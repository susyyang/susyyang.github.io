创建数据库

创建数据库:tmall
并且将数据库的编码设置为utf8，便于存放中文

DROP DATABASE IF EXISTS tmall;
CREATE DATABASE tmall DEFAULT CHARACTER SET utf8;

表结构分析

一共需要用到的9张表，以及表之间的关系。接下来，对每张表做介绍，并讲解表与表之间的关系：

这里插入一张表结构图，

各个表的介绍

表名 中文含义 介绍
Category 分类表 存放分类信息，如女装，平板电视，沙发等
Property 属性表 存放属性信息，如颜色，重量，品牌，厂商，型号等
Product 产品表 存放产品信息，如LED40EC平板电视机，海尔EC6005热水器
PropertyValue 属性值表 存放属性值信息，如重量是900g,颜色是粉红色
ProductImage 产品图片表 存放产品图片信息，如产品页显示的5个图片
Review 评论表 存放评论信息，如买回来的蜡烛很好用，么么哒
User 用户表 存放用户信息，如斩手狗，千手小粉红
Order 订单表 存放订单信息，包括邮寄地址，电话号码等信息
OrderItem 订单项列表 存放订单项信息，包括购买产品种类，数量等

正确的分析表与表之间的关系的方法

表与表之间的关系，基本上是3种
一对一
一对多
多对一
那么怎么判断关系到底是什么呢？
以分类和产品为例子，分类表里有多条数据，产品表里也有多条数据，那么他们的关系是多对多吗？ 从经验上讲，我们知道这个结果是不正确的。
正确的分析表与表之间的关系的方法是这样：
一个分类对应 多个 产品
一个产品对应 一个 分类
所以分类和产品之间的关系是一对多

表关系
从最上面的表关系图中可以看到，有关系的表之间，都是一对多关系(反过来就是多对一关系)，并没有一对一关系，和多对多关系。
所以本表列出了所有的一对多关系
最后确定表关系如下：

建表create table

建表顺序

在建表过程中，会设置外键约束，所以表和表之间有依赖关系。 因此会先建立被外键指向的表，比如User,Category，然后再是其他表。

建立用户表user

CREATE TABLE user (
  id int(11) NOT NULL AUTO_INCREMENT,
  name varchar(255) DEFAULT NULL,
  password varchar(255) DEFAULT NULL,
  PRIMARY KEY (id) --设置主键
) ENGINE=InnoDB DEFAULT CHARSET=utf8;

建立分类表category

CREATE TABLE category (
  id int(11) NOT NULL AUTO_INCREMENT,
  name varchar(255) DEFAULT NULL, --用户名
  PRIMARY KEY (id) --设置主键
) ENGINE=InnoDB  DEFAULT CHARSET=utf8;

建立属性表
注意：外键约束

CREATE TABLE property (
  id int(11) NOT NULL AUTO_INCREMENT,
  cid int(11) DEFAULT NULL,--属性所属的分类的id，category id
  name varchar(255) DEFAULT NULL,
  PRIMARY KEY (id),
  CONSTRAINT fk_property_category FOREIGN KEY (cid) REFERENCES category (id) --绑定外键
) ENGINE=InnoDB DEFAULT CHARSET=utf8;

建立产品表
注意：外键约束

CREATE TABLE product (

  id int(11) NOT NULL AUTO_INCREMENT,
  name varchar(255) DEFAULT NULL, --名字
  subTitle varchar(255) DEFAULT NULL, --标题
  orignalPrice float DEFAULT NULL, --原价
  promotePrice float DEFAULT NULL, --售价，现价
  stock int(11) DEFAULT NULL, --库存
  cid int(11) DEFAULT NULL, --产品所属分类
  createDate datetime DEFAULT NULL, --创建日期
  PRIMARY KEY (id), --id主键
  CONSTRAINT fk_product_category FOREIGN KEY (cid) REFERENCES category (id) --绑定
) ENGINE=InnoDB  DEFAULT CHARSET=utf8;

建立属性值表
注意：两个外键

CREATE TABLE propertyvalue (
  id int(11) NOT NULL AUTO_INCREMENT, --id
  pid int(11) DEFAULT NULL, --属性值 所属 属性表的id
  ptid int(11) DEFAULT NULL, --属性值 所属 产品表的id
  value varchar(255) DEFAULT NULL, --属性值
  PRIMARY KEY (id),--id主键
  CONSTRAINT fk_propertyvalue_property FOREIGN KEY (ptid) REFERENCES property (id),--绑定
  CONSTRAINT fk_propertyvalue_product FOREIGN KEY (pid) REFERENCES product (id)--绑定
) ENGINE=InnoDB  DEFAULT CHARSET=utf8;

建立产品图片表
type表示类型，产品图片分单个图片和详情图片两种
本表的外键pid，指向产品表的id字段

CREATE TABLE productimage (
  id int(11) NOT NULL AUTO_INCREMENT,--id
  pid int(11) DEFAULT NULL,--产品图片 所属 产品的id
  type varchar(255) DEFAULT NULL, --类型
  PRIMARY KEY (id),--id主键
  CONSTRAINT fk_productimage_product FOREIGN KEY (pid) REFERENCES product (id)--绑定
) ENGINE=InnoDB DEFAULT CHARSET=utf8;

评价表
外键pid，指向产品表的id字段
外键uid，指向用户表的id字段

CREATE TABLE review (
  id int(11) NOT NULL AUTO_INCREMENT,--id
  content varchar(4000) DEFAULT NULL,--评论的内容
  uid int(11) DEFAULT NULL,--评论 所属 用户的id
  pid int(11) DEFAULT NULL,--评论 所属 产品的id
  createDate datetime DEFAULT NULL, --评论创建时间
  PRIMARY KEY (id), --id主键
  CONSTRAINT fk_review_product FOREIGN KEY (pid) REFERENCES product (id),--绑定外键约束
  CONSTRAINT fk_review_user FOREIGN KEY (uid) REFERENCES user (id) --绑定外键约束
) ENGINE=InnoDB  DEFAULT CHARSET=utf8;

建立订单表，字段比较多

CREATE TABLE order_ (
  id int(11) NOT NULL AUTO_INCREMENT,--id
  orderCode varchar(255) DEFAULT NULL,--订单号
  address varchar(255) DEFAULT NULL,--收货地址
  post varchar(255) DEFAULT NULL,--邮编
  receiver varchar(255) DEFAULT NULL,--收件人
  mobile varchar(255) DEFAULT NULL,--联系手机
  userMessage varchar(255) DEFAULT NULL,--订单留言message
  createDate datetime DEFAULT NULL,--订单创建日期
  payDate datetime DEFAULT NULL,--支付日期
  deliveryDate datetime DEFAULT NULL,--发货日期
  confirmDate datetime DEFAULT NULL,--确认收货日期
  uid int(11) DEFAULT NULL,--订单所属用户的id
  status varchar(255) DEFAULT NULL,--订单状态
  PRIMARY KEY (id),--id主键
  CONSTRAINT fk_order_user FOREIGN KEY (uid) REFERENCES user (id)--绑定外键约束
) ENGINE=InnoDB  DEFAULT CHARSET=utf8;

建立订单项表

CREATE TABLE orderitem (
  id int(11) NOT NULL AUTO_INCREMENT,--id
  pid int(11) DEFAULT NULL,--该订单项 所属 产品的id
  oid int(11) DEFAULT NULL,--该订单项，所属 订单的id
  uid int(11) DEFAULT NULL,---该订单项，所属 用户的id
  number int(11) DEFAULT NULL,--该订单项的数量，比如买了5台苹果6，x土豪。
  PRIMARY KEY (id),--id主键
  CONSTRAINT fk_orderitem_user FOREIGN KEY (uid) REFERENCES user (id),--绑定
  CONSTRAINT fk_orderitem_product FOREIGN KEY (pid) REFERENCES product (id)--绑定
) ENGINE=InnoDB  DEFAULT CHARSET=utf8;
